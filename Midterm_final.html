library(RColorBrewer)
nb.cols <- 17
mycolors <- colorRampPalette(brewer.pal(8, "Set1"))(nb.cols)

m_ind_E <- merge(x = d4_ind, y = d1_E, by = "ASD_patients_with_DE_signal") %>%
  mutate(Cause_of_death = ifelse(grepl('Sudden', Cause_of_death), 'SUD', 'No_SUD')) %>%
  ggplot(aes(Cell_type, n)) +
  geom_bar(stat="identity", aes(fill = Cell_type)) +
  scale_fill_manual(values = mycolors) +
  facet_wrap(Cause_of_death~ASD_patients_with_DE_signal) +
  scale_y_continuous(trans = "log10") +
  theme(axis.text.x = element_text(angle=65, vjust = 0.6, size = 5), strip.text.x = element_text(size = 4), axis.title = element_text(size = 4)) +
  labs(title = "ASD DEGs in each Cell-type",
       subtitle = "Classified by Cause of Death: Sudden Unexpected Death (SUD)",
       x = "Cell_type",
       y = "# of DEGs (log10)")
